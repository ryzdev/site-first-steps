Transform on mouse leave

            <!DOCTYPE html>
            <html>
            <head lang="en">
                <meta charset="utf-8">
                <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.2.3/angular.min.js"></script>
                <script type="text/javascript" src="directive mouse bind.js"></script>
            </head>
            <body>
            
            <div ng-app="mouseBindApp" class="content">
                <div ng-controller="thisCtrl">
                    <h2 enter="alertMessage()"> Rollover for a popup!</h2>
                </div>

                <br>

                <h2 leave id="changeID">Transform on mouse <u>leave</u></h2>
            </div>
            
            </body>
            </html>
        
            var app = angular.module("mouseBindApp", []);

            app.controller("thisCtrl", function($scope) {
                $scope.alertMessage = function() {
                    alert("TA DAH!!!");
                }
            })

            app.directive("enter", function() {
                return function(scope, element, attrs) {
                    element.bind("mouseenter", function() {
                        scope.$apply(attrs.enter)
                    })
                }
            })

            app.directive("leave", function() {
                return function(scope, element, attrs) {
                    element.bind("mouseleave", function() {
                        document.getElementById("changeID").innerHTML="Alakazam!!";
                    })
                }
            })
